4. **Send handler** — Add handleSendEmail function:
   ```typescript
   const handleSendEmail = async () => {
     if (!summary || meetingInfo.recipientEmails.length === 0) return;
     setEmailSendState('loading');
     setEmailSendMessage('');
     try {
       const res = await authFetch('/email/send-minutes', {
         method: 'POST',
         body: JSON.stringify({
           subject: emailSubject || defaultEmailSubject,
           recipients: meetingInfo.recipientEmails,
           meetingInfo: {
             companyName: meetingInfo.companyName,
             companyAddress: meetingInfo.companyAddress,
             meetingDatetime: meetingInfo.meetingDatetime,
             meetingLocation: meetingInfo.meetingLocation,
             participants: meetingInfo.participants,
           },
           minutesMarkdown: summary,
         }),
       });
       if (res.ok) {
         const data = await res.json();
         setEmailSendState('success');
         setEmailSendMessage(`Da gui thanh cong den ${data.count} dia chi`);
       } else {
         const data = await res.json().catch(() => ({ error: 'Gui that bai' }));
         setEmailSendState('error');
         setEmailSendMessage(data.error || 'Gui that bai. Vui long thu lai.');
       }
     } catch {
       setEmailSendState('error');
       setEmailSendMessage('Gui that bai. Vui long thu lai.');
     }
   };
   ```
